    def computer_error_for_give_point(w, b, points): # 计算出 观测值与计算值 之间的误差， 并累加,最后返回 平均误差
    loss = 0
    for i in range(len(points)):
    x = points[i, 0]
    y = points[i, 1]
    loss += ((w * x + b ) - y) ** 2
    return loss/float(len(points))
