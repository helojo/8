    1.random模块常用方法：

    import random

#生成0-1之间的伪浮点随机数（默认）
print(random.random())

#生成2.0-5.0之间的伪浮点随机数
print(random.uniform(2.0,5.0))

#生成1-10之间的伪整数随机数
print(random.randint(1,10))

#设置种子
random.seed(11)
print(random.random())

print(random.random())

random.seed(11)
