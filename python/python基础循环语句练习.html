    1、使用while循环输入 1 2 3 4 5 6     8 9 10

    n = 1
while n < 11:
    if n == 7:
        pass
    else:
        print(n)
    n = n + 1

    2、求1-100的所有数的和

    n = 1
s = 0
while n < 101:
    s = s + n
    n = n + 1
print(s)

    3、输出 1-100 内的所有奇数

    n = 1
while n < 101:
    temp = n % 2
    if temp == 1:
        print(n)
    else:
        pass
    n = n + 1

    4、输出 1-100 内的所有偶数

    n = 1
while n < 101:
    temp = n % 2
    if temp == 0:
        print(n)
    else:
        pass
    n = n + 1

    5、求1-2+3-4+5 ... 99的所有数的和

    n = 1
s = 0
while n < 101:
    temp = n % 2
    if temp == 0:
        s = s - n
    else:
        s = s + n
    n = n + 1
print(s)

    7、用户登陆（三次机会重试）

    count = 0
while count < 3:
    user = input('>>>')
    pwd = input('>>>')
    if user == 'alex' and pwd == '123':
        print('欢迎登陆')
        break
    else:
        print('用户名或者密码错误')
    count = count + 1
