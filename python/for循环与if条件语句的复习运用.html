    主要是分析思路，水仙花数的特点就是每位数的3次幂的结果相加的和等于这个数字。那就要考虑如何将这个数的每位的数字提出取来
    def waterFlower():
    for i in range(100, 1000):
        a = i // 100        #取百位数字
        b = i % 100 // 10   #取十位数字
        c = i % 10          #取个位数字
        if a ** 3 + b ** 3 + c ** 3 == i:
            print(i)
